Encontrar fatores primos de um número

Retorna uma lista com os fatores primos de um inteiro N.


def prime_factors(N):
    #Find prime factors
    outFactors =[]
    
    i = 2
    
    while i**2 < N:
        if (N % i ==0):
            outFactors.append(i)            
            N /= i
        else:
            i += 1
        
    if (N>1):
        outFactors.append(N)
    
    return(outFactors)
