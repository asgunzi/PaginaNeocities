Fibonacci:
def Fibo(nTermos):
    fibA=1
    fibB=1
    
    listFib =[fibA, fibB]
    
    while len(listFib) < nTermos:
        fibC = fibA + fibB
        listFib.append(fibC)
        
        fibA = fibB
        fibB = fibC
    return(listFib)
